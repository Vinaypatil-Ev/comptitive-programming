# 133.Clone Graph

### 1.BFS approach

```py
"""
# Definition for a Node.
class Node:
    def __init__(self, val = 0, neighbors = None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []
"""

class Solution:
    def cloneGraph(self, node: 'Node') -> 'Node':
        if not node:
            return node

        d[node] = cg
        q = collections.deque()
        q.append(node)
        while q:
            cnode = q.popleft()
            cgnode = d[cnode]
            for adjnode in cnode.neighbors:
                cgadj = Node(adjnode.val)
                if not adjnode in d:
                    d[adjnode] = cgadj
                    q.append(adjnode)
                cgnode.neighbors.append(d[adjnode])
    
        return cg
```

Time complexity - `O(n)`
Space comlexity - `O(n)`

### 2.DFS approach

```py
"""
# Definition for a Node.
class Node:
    def __init__(self, val = 0, neighbors = None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []
"""

class Solution:
    def cloneGraph(self, node):
        if node is None: 
            return None
        d = {}
        def dfs(node):
            if node in d: 
                return d[node]
            cg = Node(node.val, [])
            d[node] = cg
            for adjnode in node.neighbors:
                cg.neighbors.append(dfs(adjnode))
            return cg
  
        return dfs(node)
```

Time complexity - `O(n + m)` n: no of node m: no of edges
Space comlexity - `O(n)`
