# 133.Clone Graph

### 1.BFS approach

```py
"""
# Definition for a Node.
class Node:
    def __init__(self, val = 0, neighbors = None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []
"""

class Solution:
    def cloneGraph(self, node: 'Node') -> 'Node':
        if not node:
            return node
        cg = Node(node.val)
        d = {}
        d[node] = cg
        q = collections.deque()
        q.append(node)
        while q:
            cnode = q.popleft()
            cgnode = d[cnode]
            for adjnode in cnode.neighbors:
                cgadj = Node(adjnode.val)
                if not adjnode in d:
                    d[adjnode] = cgadj
                    q.append(adjnode)
                cgnode.neighbors.append(d[adjnode])
                
        return cg
```
