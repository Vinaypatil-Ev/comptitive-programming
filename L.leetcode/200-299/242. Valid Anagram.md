# [242.Valid anagrams]()

### 1.Rolling hash approach

```py
class Solution:
    def isAnagram(self, s: str, t: str) -> bool:
        if len(s) != len(t):
            return False
        d = {}
        for i in range(len(s)):
            d[s[i]] = d.get(s[i], 0) + 1
        for i in range(len(t)):
            x = d.get(t[i], 0)
            if x == 0:
                return False
            if x == 1:
                d.pop(t[i])
            elif x > 1:
                d[t[i]] -= 1
        if not d:
            return True
        return False
```
